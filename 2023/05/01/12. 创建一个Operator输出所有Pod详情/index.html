Operator读取所有的Pod，Deployment,Service

倪克松 2023-05-01 20:25:26
Categories： Kubernetes学习笔记 Tags：

后面的实战中，咱们的SpringBoot应用会以Pod的形式运行在K8S环境，通过client-java的API操作K8S，而K8S自然不会允许Pod随意对环境中的pod、deployment这些资源进行操作，因此我们要遵照K8S的规范进行RBAC相关的操作；
K8S的角色设置可以非常详细，但也会更复杂，咱们还是不要花太多时间在这上面吧，这里我选择了使用K8S自带的最高权限的角色：cluster-admin，您可以按照自己的实际情况定制角色，下面是具体操作；

编写代码

package com.nike;

import com.google.gson.GsonBuilder;
import io.kubernetes.client.openapi.ApiClient;
import io.kubernetes.client.openapi.Configuration;
import io.kubernetes.client.openapi.apis.AppsV1Api;
import io.kubernetes.client.openapi.apis.CoreV1Api;
import io.kubernetes.client.openapi.models.V1DeploymentList;
import io.kubernetes.client.openapi.models.V1PodList;
import io.kubernetes.client.openapi.models.V1ServiceList;
import io.kubernetes.client.util.ClientBuilder;
import io.kubernetes.client.util.KubeConfig;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.io.FileNotFoundException;
import java.io.FileReader;

@SpringBootApplication
public class OutsideClusterApplication {

    public static void main(String[] args) throws Exception {
        // 存放K8S的config文件的全路径
        String kubeConfigPath = "/root/.kube/config";

        // 以config作为入参创建的client对象，可以访问到K8S的API Server
        ApiClient client = ClientBuilder
                .kubeconfig(KubeConfig.loadKubeConfig(new FileReader(kubeConfigPath)))
                .build();

        Configuration.setDefaultApiClient(client);

        CoreV1Api api = new CoreV1Api();

        // 调用客户端API取得所有pod信息
        V1PodList v1PodList = api.listPodForAllNamespaces(null, null, null, null, null, null, null, null, null);
        // 调用客户端API取得所有Service信息
        V1ServiceList v1ServiceList = api.listServiceForAllNamespaces(null, null, null, null, null, null, null, null, null);
        // 创建AppsV1Api对象
        AppsV1Api api1 = new AppsV1Api();
        // 列出所有namespace中的Deployment
        V1DeploymentList deploymentList = api1.listDeploymentForAllNamespaces(null, null, null, null, null, null, null, null, null);
        // 使用jackson将集合对象序列化成JSON，在日志中打印出来
        // System.out.println("pod info { " + new GsonBuilder().setPrettyPrinting().create().toJson(v1PodList)+"}");
        // System.out.println("Service info { " + new GsonBuilder().setPrettyPrinting().create().toJson(v1ServiceList)+"}");
        System.out.println("Deployment info { " + new GsonBuilder().setPrettyPrinting().create().toJson(deploymentList)+"}");
    }
}

以上是在master主机上的编写，如果不在master，则把config文件复制一份到某文件夹，并把路径配给kubeConfigPath
