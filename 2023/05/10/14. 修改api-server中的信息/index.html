配置Kubernetes API Server的连接参数

倪克松 2023-05-10 21:41:44
Categories： Kubernetes学习笔记 Tags： 
public  static void configApiServer() throws Exception {
    // 加载配置文件
    Properties configProps = new Properties();
    configProps.load(new FileReader("config.properties"));

    // 创建ApiClient对象
    ApiClient client = ClientBuilder.standard()
            .setBasePath(configProps.getProperty("api.server.url"))  // 设置API Server的URL
            .setVerifyingSsl(false)                                   // 禁用SSL证书验证
            .setAuthentication(new ClientCertificateAuthentication(  // 设置客户端证书和私钥
                    readPrivateKey(configProps.getProperty("client.cert.private.key")),
                    readCertificate(configProps.getProperty("client.cert.public.key"))
            ))
            .build();

    // 根据ApiClient对象创建Configuration对象
    Configuration.setDefaultApiClient(client);
}
private static byte[] readPrivateKey(String privateKeyFile) throws Exception {
    return Files.readAllBytes(Paths.get(privateKeyFile));
}

private static byte[] readCertificate(String certificateFile) throws Exception {
    return Files.readAllBytes(Paths.get(certificateFile));
}
